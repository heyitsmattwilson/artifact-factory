-- Nielsen Category Context — Artifact Factory -- Replace :category_name with the category (e.g. 'Salty Snacks') -- SOT: main.niq.vw_niq_cpg_joined SELECT CONCAT('$', ROUND(SUM(n.dollars) / 1e9, 1), 'B') AS niq_dollars, CONCAT(ROUND((SUM(n.dollars) / NULLIF(SUM(n.dollars_ya),0) - 1)*100, 1), '%') AS niq_yoy, CONCAT(ROUND(SUM(n.any_promo_dollars)/NULLIF(SUM(n.dollars),0)*100, 1), '%') AS niq_promo, CONCAT('$', ROUND(AVG(n.avg_unit_price), 2)) AS niq_price, MAX(n.period_description) AS niq_period FROM main.niq.vw_niq_cpg_joined n JOIN main.catalog.vw_fact_products p ON n.global_product_event_id = p.global_product_event_id WHERE p.category_name ILIKE :category_name AND n.day = (SELECT MAX(day) FROM main.niq.vw_niq_cpg_joined) LIMIT 1;
